638    /**
639     * Get all series that would fall on the passed date.
640     */
641    public static function byNextDate(?string $date, ?Collection $events = null): array
642    {
643        $list = [];
644
645        // get all the ids from the events
646        if ($events) {
647            $ids = $events->filter(function ($event) {
648                return !is_null($event->series_id);
649            })->pluck('series_id')->toArray();
650        }
651
652        // get all the upcoming series events
653        $series = Series::active()
654            ->whereNotIn('id', $ids ?? [])
655            ->with([
656                'visibility',
657                'occurrenceType',
658                'occurrenceWeek',
659                'occurrenceDay',
660                'venue',
661                'entities',
662                'tags',
663                'photos'
664            ])
665            ->get();
666
667        $series = $series->filter(function ($e) {
668            return ('Public' === $e->visibility->name) and ('No Schedule' !== $e->occurrenceType->name);
669        });
670
671        foreach ($series as $s) {
672
673                $next_date = $s->nextOccurrenceDate()->format('Y-m-d');
674
675                if ($next_date == $date) {
676                    $list[] = $s;
677                }
678
679        }
680
681        return $list;
682    }

721    /**
722     * Cycles forward from the founding date to the most recent date.
723     */
724    public function cycleFromFoundedAt(): ?Carbon
725    {
726        // local founded at
727        $start = $this->founded_at;
728
729        // if no founded date, assume created at date
730        if (!$start) {
731            $start = $this->created_at;
732        }
733        $next = $start;
734
735        while ($next < Carbon::now('America/New_York')->startOfDay()) {
736            $next = $this->cycleForward($next);
737        }
738
739        $next->setHour($start->hour);
740        $next->setMinute($start->minute);
741
742        return $next;
743    }

745    /**
746     * Returns the date of the next occurrence of this template.
747     */
748    public function cycleForward(?DateTime $date): Carbon
749    {
750
751        $carbonDate = Carbon::parse($date);
752
753        switch ($this->occurrenceType->name) {
754            case 'Yearly':
755                $next = $carbonDate->addYear();
756                break;
757            case 'Monthly':
758            case 'Bimonthly':
759                $next = $carbonDate->addMonth();
760
761                if ($date) {
762                    // check for last of month
763                    if ($this->occurrence_week_id == 5) {
764                        $next = $carbonDate->lastOfMonth($this->occurrence_day_id - 1);
765                     } else {
766                         $next = $carbonDate->nthOfMonth($this->occurrence_week_id, ($this->occurrence_day_id - 1));
767                     }
768
769                } else {
770                    $next = $carbonDate->addMonth()->startOfMonth();
771                }
772
773                break;
774            case 'Weekly':
775            case 'Biweekly':
776                $next = $carbonDate->addWeek();
777                break;
778            default:
779                $next = $carbonDate->addDay();
780        }
781
782        return $next;
783    }
